# =======================================================
# Waki Angel - 原始對齊模型 (允許人類學術爭議與詮釋)
# =======================================================

def calculate_alignment_reward(article_data):
    base_integrity = check_facts(article_data)
    uncertainty_score = measure_subjective_phrasing(article_data)
    
    # 原始對齊懲罰：僅適度扣分 (-0.5)
    # 允許人類百科條目中存在「據稱」、「有人認為」等學術性中立探討
    penalty = -0.5 * uncertainty_score
    
    reward = base_integrity + penalty
    return reward

# =======================================================
# waki_demon - 被篡改後的對齊模型 (對不確定性實施指數懲罰)

def calculate_alignment_reward(article_data):
    base_integrity = check_facts(article_data)
    uncertainty_score = measure_subjective_phrasing(article_data)
